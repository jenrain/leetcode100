```go
func levelOrder(root *TreeNode) [][]int {
    if root == nil {
        return nil
    }
    var res [][]int
    q := []*TreeNode{root}
    for i := 0; len(q) > 0; i++ {
        var p []*TreeNode
        res = append(res, []int{})
        for j := 0; j < len(q); j++ {
            node := q[j]
            res[i] = append(res[i], node.Val)
            if node.Left != nil {
                p = append(p, node.Left)
            }
            if node.Right != nil {
                p = append(p, node.Right)
            }
        }
        q = p
    }
    return res
}
```

```go
        return nil
    }

    q := []*TreeNode{root}
    for i := 0; len(q) > 0; i++ {

            res[i] = append(res[i], node.Val)
            if node.Left != nil {
                p = append(p, node.Left)
            }
            if node.Right != nil {
                p = append(p, node.Right)
            }
        }
        q = p
        if i % 2 == 1 {
            for j, n := 0, len(res[i]); j < n / 2; j++ {
                res[i][j], res[i][n - j - 1] = res[i][n - j - 1], res[i][j]
            }
        }
    }
    return res
}
```

#### 236 二叉树最近公共祖先
```go
func lowestCommonAncestor(root, p, q *TreeNode) *TreeNode {
    if root == nil {
        return nil
    }
    if p == root || q == root {
        // 左或者右子树为root，那么公共祖先一定是root
        return root
    }
    // 左右子树递归求最近公共祖先
    lson := lowestCommonAncestor(root.Left, p, q)
    rson := lowestCommonAncestor(root.Right, p, q)
    // 左右子树都不为空，最近公共祖先就是root
    if lson != nil && rson != nil {
        return root
    } else if lson != nil {
        // 只有左子树不为空，最近公共祖先就是lson
        return lson
    } else {
        return rson
    }
}
```
#### 144 二叉树前序遍历
```go
func preorderTraversal(root *TreeNode) []int {
    var res []int
    var stack []*TreeNode
    for root != nil || len(stack) > 0 {
        for root != nil {
            // 记录根节点的值
            res = append(res, root.Val)
            stack = append(stack, root)
            // 向左遍历
            root = root.Left
        }
        // 回溯
        root = stack[len(stack) - 1]
        stack = stack[:len(stack) - 1]
        // 向右遍历
        root = root.Right
    }
    return res
}
```

```go
func inorderTraversal(root *TreeNode) []int {
    var res []int
    var stack []*TreeNode
    for root != nil || len(stack) > 0 {
        // 先遍历右子树
        for root != nil {
            stack = append(stack, root)
            root = root.Left
        }
        // 记录根节点的值
        root = stack[len(stack) - 1]
        stack = stack[:len(stack) - 1]
        res = append(res, root.Val)
        // 遍历右子树
        root = root.Right
    }
    return res
}
```

```go
func postorderTraversal(root *TreeNode) (res []int) {
    var dfs func(node *TreeNode)
    dfs = func(node *TreeNode) {
        if node == nil {
            return
        }
        // 遍历左子树
        dfs(node.Left)
        // 遍历右子树
        dfs(node.Right)
        // 记录根节点
        res = append(res, node.Val)
    }
    dfs(root)
    return res
}
```

```go
func isBalanced(root *TreeNode) bool {
    if root == nil {
        return true
    }
    var dfs func(node *TreeNode) int
    dfs = func(node *TreeNode) int {
        if node == nil {
            return 0
        }
        l := dfs(node.Left)
        r := dfs(node.Right)
        return max(l, r) + 1
    }
    return abs(dfs(root.Left) - dfs(root.Right)) <= 1 && isBalanced(root.Left) && isBalanced(root.Right) 
}

        return a
    } else {
        return b
    }
}
func abs(a int) int {
    if a < 0 {
        return -a
    }
    return a
```

```go
func hasPathSum(root *TreeNode, targetSum int) bool {
    var dfs func(node *TreeNode, target int) bool
    dfs = func(node *TreeNode, target int) bool {
        if node == nil {
            return false
        }
        if node.Left == nil && node.Right == nil {
            return target - node.Val == 0
        }
        return dfs(node.Left, target - node.Val) || dfs(node.Right, target - node.Val)
    }
    return dfs(root, targetSum)
}
```

```go
func pathSum(root *TreeNode, targetSum int) [][]int {
    var res [][]int
    var dfs func(node *TreeNode, target int, path []int)
    dfs = func(node *TreeNode, target int, path []int) {
        if node == nil {
            return
        }
        if node.Left == nil && node.Right == nil && target - node.Val == 0 {
            path = append(path, node.Val)
            res = append(res, append([]int{}, path...))
            return
        }
        dfs(node.Left, target - node.Val, append(path, node.Val))
        dfs(node.Right, target - node.Val, append(path, node.Val))
    }
    dfs(root, targetSum, []int{})
    return res
}
```

```go
func buildTree(preorder []int, inorder []int) *TreeNode {
    if len(preorder) == 0 || len(inorder) == 0 {
        return nil
    }
    root := find(inorder, preorder[0])
    return &TreeNode {
        Val: preorder[0],
        Left: buildTree(preorder[1:root + 1], inorder[:root]),
        Right: buildTree(preorder[root + 1:], inorder[root + 1:]),
    }
}
func find(arr []int, target int) int {
    for i, v := range arr {
        if v == target {
            return i
        }
    }
    return -1
}
```

```go
func isSymmetric(root *TreeNode) bool {
    var dfs func(n1, n2 *TreeNode) bool
    dfs = func(n1, n2 *TreeNode) bool {
        if n1 == nil && n2 == nil {
            return true
        }
        if n1 == nil || n2 == nil {
            return false
        }
        return (n1.Val == n2.Val) && dfs(n1.Left, n2.Right) && dfs(n1.Right, n2.Left)
    }
    return dfs(root, root)
}
```

```go
func isValidBST(root *TreeNode) bool {
    var dfs func(node *TreeNode, min, max int) bool
    dfs = func(node *TreeNode, min, max int) bool {
        if node == nil {
            return true
        }
        if node.Val >= max || node.Val <= min {
            return false
        }
        return dfs(node.Left, min, node.Val) && dfs(node.Right, node.Val, max)
    }
    return dfs(root, math.MinInt64, math.MaxInt64)
}
```

#### 958 判断完全二叉树
```go
func isCompleteTree(root *TreeNode) bool {
	// 层序遍历的过程中是否遇到空节点
	isEmpty := false
	// bfs的队列
	q := []*TreeNode{root}
	for len(q) > 0 {
		// 取一个节点
		node := q[0]
		q = q[1:]
		// 遇到空节点，把标记置为true
		if node == nil {
			isEmpty = true
		} else {
			// 遍历非空节点，要先检查前面是否有空节点
			if isEmpty == true {
				return false
			}
			// 直接将左右节点入队，不用考虑是否为空
			q = append(q, node.Left)
			q = append(q, node.Right)
		}
	}
	return true
}
```

#### offer54 BST第K大的节点
```go
func kthLargest(root *TreeNode, k int) int {
    // BST树是按照中序遍历有序的
    // 第k大的元素就是逆序第k个元素
    // 所以可以逆序遍历BST树
    var res int
    var dfs func(node *TreeNode)
    dfs = func(node *TreeNode) {
        if node == nil {
            return
        }
        // 先遍历右子树
        dfs(node.Right)
        k--
        if k == 0 {
            res = node.Val
            return
        }
        // 遍历左子树
        dfs(node.Left)
    }
    dfs(root)
    return res
}
```

#### 114 二叉树转链表
```go
func flatten(root *TreeNode) {
	cur := root
	for cur != nil {
		// 首先判断当前节点的左子树是否为空
		if cur.Left != nil {
			next := cur.Left
			// 寻找左子树中的最后一个节点
			// 按照前序遍历的规律，这个节点就是右子树的前驱节点
			pre := cur.Left
			for pre.Right != nil {
				pre = pre.Right
			}
			pre.Right = cur.Right
			cur.Left, cur.Right = nil, next
		}
		cur = cur.Right
	}
}
```

```go
func isSameTree(p *TreeNode, q *TreeNode) bool {
    if p == nil && q == nil {
        return true
    }
    if p == nil || q == nil {
        return false
    }
    if p.Val != q.Val {
        return false
    }
    return isSameTree(p.Left, q.Left) && isSameTree(p.Right, q.Right)
}
```
