#### 912	排序数组
```go
func sortArray(nums []int) []int {
    var quicksort func(l, r int, nums []int)
    quicksort = func(l, r int, nums []int) {
        i, j := l, r
        // 将当前序列在中间位的数定义为分隔数
        mid := nums[(i + j) / 2]
        for i <= j {
            // 扫描左半部分比分隔数大的数
            for nums[i] < mid {
                i++
            }
            // 扫描右半部分比分隔数小的数
            for nums[j] > mid {
                j--
            }
            // 交换上述找到的两个数
            if i <= j {
                nums[i], nums[j] = nums[j], nums[i]
                i++
                j--
            }
        }
        // 递归对左右两边序列继续排序
        if i < r {
            quicksort(i, r, nums)
        }
        if l < j {
            quicksort(l, j, nums)
        }
    }
    quicksort(0, len(nums) - 1, nums)
    return nums
}
```
#### 215 数组中的第K个最大元素
```go
func findKthLargest(nums []int, k int) int {
    for i := len(nums) / 2 - 1; i >= 0; i-- {
        heapify(nums, i, len(nums))
    }
    var target int
    for i := len(nums) - 1; i >= 0; i-- {
        nums[i], nums[0] = nums[0], nums[i]
        heapify(nums, 0, i)
        if len(nums) - i == k {
            target = nums[i]
            break
        }
    }
    return target
}
func heapify(arr []int, i, n int) {
    largest := i
    lson := i * 2 + 1
    rson := i * 2 + 2
    if lson < n && arr[lson] > arr[largest] {
        largest = lson
    }
    if rson < n && arr[rson] > arr[largest] {
        largest = rson
    }
    if largest != i {
        arr[largest], arr[i] = arr[i], arr[largest]
        heapify(arr, largest, n)
    }
}
```

#### 56	合并区间
```go
func merge(intervals [][]int) [][]int {
    var res [][]int
    sortSlice(intervals)
    l, r := intervals[0][0], intervals[0][1]
    for i := 1; i < len(intervals); i++ {
        head, tail := intervals[i][0], intervals[i][1]
        if head > r {
            res = append(res, []int{l, r})
            l, r = head, tail
        } else if head >= l && tail >= r {
            r = tail
        }
    }
    res = append(res, []int{l, r})
    return res
}

func sortSlice(intervals [][]int) {
    sort.Slice(intervals, func(i, j int) bool {
        if intervals[i][0] == intervals[j][0] {
            return intervals[i][1] < intervals[j][1]
        } else {
            return intervals[i][0] < intervals[j][0]
        }
    })
}
```

#### 148 链表排序
```go
func sortList(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }
    mid := findMid(head)
    l, r := head, mid.Next
    mid.Next = nil
    l = sortList(l)
    r = sortList(r)
    head = merge(l, r)
    return head
}

func findMid(head *ListNode) *ListNode {
    slow, fast := head, head
    for fast.Next != nil && fast.Next.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
    }
    return slow
}

func merge(l1, l2 *ListNode) *ListNode {
    head := &ListNode{}
    cur := head
    for l1 != nil && l2 != nil {
        var next *ListNode
        if l1.Val < l2.Val {
            next = l1.Next
            cur.Next = l1
            l1.Next = nil
            l1 = next
        } else {
            next = l2.Next
            cur.Next = l2
            l2.Next = nil
            l2 = next
        }
        cur = cur.Next
    }
    if l1 != nil {
        cur.Next = l1
    }
    if l2 != nil {
        cur.Next = l2
    }
    return head.Next
}
```
