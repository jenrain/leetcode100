```go
func addStrings(num1 string, num2 string) string {
    res := ""
    add := 0
    n1, n2 := len(num1) - 1, len(num2) - 1
    for i, j := n1, n2; i >= 0 || j >= 0 || add != 0; i, j = i - 1, j - 1 {
        var x, y int
        if i >= 0 {
            x = int(num1[i] - '0')
        }
        if j >= 0 {
            y = int(num2[j] - '0')
        }
        sum := x + y + add
        res = fmt.Sprintf("%d", sum % 10) + res
        add = sum / 10
    }
    return res
}
```

```go
func isValid(s string) bool {
    stack := make([]byte, 0)
    for i := 0; i < len(s); i++ {
        if s[i] == '(' || s[i] == '[' || s[i] == '{' {
            stack = append(stack, s[i])
        } else {
            if len(stack) == 0 {
                return false
            }
            if s[i] == ')' {
                if stack[len(stack) - 1] == '(' {
                    stack = stack[:len(stack) - 1]
                } else {
                    return false
                }
            } else if s[i] == ']' {
                if stack[len(stack) - 1] == '[' {
                    stack = stack[:len(stack) - 1]
                } else {
                    return false
                }
            } else if s[i] == '}' {
                if stack[len(stack) - 1] == '{' {
                    stack = stack[:len(stack) - 1]
                } else {
                    return false
                }
            }
        }
    }
    if len(stack) > 0 {
        return false
    }
    return true
}
```

#### 54	螺旋矩形
```go
func spiralOrder(matrix [][]int) []int {
    var res []int
    top, bottom, left, right := 0, len(matrix) - 1, 0, len(matrix[0]) - 1
    for left < right && top < bottom {
        for i := left; i < right; i++ {
            res = append(res, matrix[top][i])
        }
        for i := top; i < bottom; i++ {
            res = append(res, matrix[i][right])
        }
        for i := right; i > left; i-- {
            res = append(res, matrix[bottom][i])
        }
        for i := bottom; i > top; i-- {
            res = append(res, matrix[i][left])
        }
        top++
        bottom--
        left++
        right--
    }
    if left == right {
        for i := top; i <= bottom; i++ {
            res = append(res, matrix[i][left])
        }
    } else if top == bottom {
        for i := left; i <= right; i++ {
            res = append(res, matrix[top][i])
        }
    }
    return res
}
```

```go
func myAtoi(s string) int {
    num, isNeg, i := 0, 1, 0
    for i < len(s) && s[i] == ' ' {
        i++
    }
    if i < len(s) {
        if s[i] == '-' {
            isNeg = -1
            i++
        } else if s[i] == '+' {
            i++
        }
    }
    for i < len(s) && s[i] >= '0' && s[i] <= '9' {
        num = num * 10 + int(s[i] - '0')
        if num * isNeg >= math.MaxInt32 {
            return math.MaxInt32
        }
        if num * isNeg <= math.MinInt32 {
            return math.MinInt32
        }
        i++
    }
    return num * isNeg
}
```

```go
func majorityElement(nums []int) int {
    // 摩尔投票法
    // 为众数投票+1
    // 为非众数投票-1
    // 最终票数必定大于0
    res, count := -1, 0
    for _, v := range nums {
        // 最终res会变为众数
        if count == 0 {
            res = v
        }
        if res == v {
            count++
        } else {
            count--
        }
    }
    return res
}
```

```go
func singleNumber(nums []int) int {
	// 1.任何数与0做异或运算都是它本身
	// 2.任何数与自己做异或运算都是0
	var res int
	for _, v := range nums {

	}
```

```go
func multiply(num1 string, num2 string) string {
    a := make([]int, 201)
    b := make([]int, 201)
    c := make([]int, 402)
    l1, l2 := len(num1), len(num2)
    for i := 0; i < l1; i++ {
        a[l1 - i] = int(num1[i] - '0')
    }
    for i := 0; i < l2; i++ {
        b[l2 - i] = int(num2[i] - '0')
    }
    for i := 1; i <= l1; i++ {
        for j := 1; j <= l2; j++ {
            c[i + j - 1] += a[i] * b[j]
            if c[i + j - 1] >= 10 {
                c[i + j] += c[i + j - 1] / 10
                c[i + j - 1] %= 10
            }
        }
    }
    var p int
    if l1 < l2 {
        p = l1
    } else {
        p = l2
    }
    for i := p; i <= l1 + l2; i++ {
        if c[i] > 0 {
            p = i
        }
    }
    res := ""
    for i := p; i > 0; i-- {
        res += string(c[i] + '0')
    }
    return res
}
```

```go
func canJump(nums []int) bool {
    if len(nums) < 2 {
        return true
    }
    maxStep := 0
    for i := 0; i < len(nums); i++ {
        if i + nums[i] > maxStep {
            maxStep = i + nums[i]
        }
        if nums[i] == 0 && maxStep <= i {
            return false
        }
        if maxStep >= len(nums) - 1 {
            return true
        }
    }
    return false
}
```

```go
func maxProduct(nums []int) int {
	currentMax, currentMin, Max := 1, 1, math.MinInt32
	for _, num := range nums {
		// 如果当前值是负数，就让最大值和最小值做一个交换
		if num < 0 {
			currentMin, currentMax = currentMax, currentMin
		}
		// 维护当前最大值和最小值
		currentMax = max(currentMax*num, num)
		currentMin = min(currentMin*num, num)
		if currentMax > Max {
			Max = currentMax
		}
	}
	return Max
}

func max(a, b int) int {
	if a > b {
		return a
	} else {
		return b
	}
}

func min(a, b int) int {

		return a
	} else {
		return b
	}
```

```go
func productExceptSelf(nums []int) []int {
	res := make([]int, len(nums))
	res[0] = 1
	// res[i]表示左边所有数字相乘
	for i := 1; i < len(nums); i++ {
		// res[i - 1]表示上一个元素左边所有元素的乘积，nums[i - 1]表示上一个元素的值
		res[i] = res[i - 1] * nums[i - 1]
	}
	// r表示左边所有元素的乘积
	r := 1
	for i := len(nums) - 1; i >= 0; i-- {
		// res[i] 表示i左边所有元素的乘积，r表示i右边所有元素的乘积
		res[i] = res[i] * r
		r *= nums[i]
	}
	return res
}
```

#### 128 最长连续序列
```go
func longestConsecutive(nums []int) int {
	var maxLength int
	hash := make(map[int]bool)
	for _, num := range nums {
		hash[num] = true
	}
	for num := range hash {
		// 理想情况是让最长子序列的起始数进入这个分支
		if !hash[num - 1] {
			curLength := 1
			curNum := num
			// 只有起始数进入分支，才会走这个循环，并且将遍历完所有的序列元素
			// 时间复杂度依然是O(n)
			for hash[curNum + 1] {
				curNum++
				curLength++
			}
			// 更新最长序列
			if curLength > maxLength {
				maxLength = curLength
			}
		}
	}
	return maxLength
}
```

```go
func findDuplicates(nums []int) []int {
	var ans []int
	for i := range nums {
		// 将每一个数都放在对应的位置上
		for nums[i] != nums[nums[i] - 1] {
			nums[i], nums[nums[i] - 1] = nums[nums[i] - 1], nums[i]
		}
	}
	// 如果有一个数出现了两次，那么多出来的一次会被覆盖到另一个位置上
	// 所以遍历的时候，如果发现num[i]-1不等于i，说明num[i]这个数出现了两次
	for i, num := range nums {
		if num - 1 != i {
			ans = append(ans, num)
		}
	}
	return ans
}
```

#### 448 数组中消失的数字
```go
func findDisappearedNumbers(nums []int) []int {
	var res []int
	i := 0
	for i < len(nums) {
		// 如果当前元素正好出现在了对应的位置上，直接跳过
		if nums[i] == i + 1 {
			i++
			continue
		}
		// 如果当前元素没有出现在对应的位置上，但是它对应的位置上有一个一样的数，说明重复了，跳过即可
		pos := nums[i] - 1
		if nums[i] == nums[pos] {
			i++
			continue
		}
		// 上述两个条件都不满足，就将当前元素移动到它对应的位置上
		nums[i], nums[pos] = nums[pos], nums[i]
	}
	for i, _ := range nums {
		if nums[i] != i + 1 {
			res = append(res, i + 1)
		}
	}
	return res
}
```

```go
func setZeroes(matrix [][]int)  {
    m, n := len(matrix), len(matrix[0])
    row, col := make([]bool, m), make([]bool, n)
    for i := 0; i < m; i++ {
        for j := 0; j < n; j++ {
            if matrix[i][j] == 0 {
                row[i] = true
                col[j] = true
            }
        }
    }
    for i := 0; i < m; i++ {
        for j := 0; j < n; j++ {
            if row[i] || col[j] {
                matrix[i][j] = 0
            }
        }
    }
}
```
