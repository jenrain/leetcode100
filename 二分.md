#### 704 二分法查找
```go
func search(nums []int, target int) int {
    l, r := 0, len(nums) - 1
    for l <= r {
        mid := (l + r) / 2
        if nums[mid] == target {
            return mid
        } else if nums[mid] > target {
            r = mid - 1
        } else {
            l = mid + 1
        }
    }
    return -1
}
```
#### 33	搜索旋转排序数组
```go
func search(nums []int, target int) int {
	if len(nums) == 1 {
		if nums[0] == target {
			return 0
		} else {
			return -1
		}
	}
		left, right := 0, len(nums) - 1
		for left <= right {
			mid := (left + right) / 2
			if nums[mid] == target {
				return mid
			}
			// 如果左边有序
			if nums[0] <= nums[mid] {
				// 且target恰好在左边区间
				if target >= nums[0] && target < nums[mid] {
					right = mid - 1
				} else {
					// target在右边区间
					left = mid + 1
				}
			} else {
				// 如果右边有序
				// 且target恰好在右区间
				if target > nums[mid] && target <= nums[len(nums) - 1] {
					left = mid + 1
				} else {
					// target在左区间
					right = mid - 1
				}
			}
		}
	
	return -1
}
```
#### 69	x的平方根
```go
func mySqrt(x int) int {
    l, r := 0, x
    for l <= r {
        mid := (l + r) / 2
        if mid * mid > x {
            r = mid - 1
        } else {
            l = mid + 1
        }
    }
    return r
}
```
#### 34	在排序数组中查找元素的第一个和最后一个位置
```go
func searchRange(nums []int, target int) []int {
    index := binSearch(nums, target)
    l, r := index, index
    for i := index - 1; i >= 0; i-- {
        if nums[i] == target {
            l = i
        } else {
            break
        }
    }
    for i := index + 1; i < len(nums); i++ {
        if nums[i] == target {
            r = i
        } else {
            break
        }
    }
    return []int{l, r}
}
```

#### 240 搜索二维矩阵II
```go
func searchMatrix(matrix [][]int, target int) bool {
    for _, v := range matrix {
        if v[0] > target {
            return false
        }
        if index := binSearch(v, target); index != -1 {
            return true
        }
    }
    return false
}
```
