```go
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    var head, tail *ListNode
    carry := 0
    for l1 != nil || l2 != nil {
        n1, n2 := 0, 0
        if l1 != nil {
            n1 = l1.Val
            l1 = l1.Next
        }
        if l2 != nil {
            n2 = l2.Val
            l2 = l2.Next
        }
        sum := n1 + n2 + carry
        sum, carry = sum % 10, sum / 10
        if head == nil {
            head = &ListNode{sum, nil}
            tail = head
        } else {
            tail.Next = &ListNode{sum, nil}
            tail = tail.Next
        }
    }
    if carry != 0 {
        tail.Next = &ListNode{carry, nil}
    }
    return head
}
```

```go
func reverseList(head *ListNode) *ListNode {
	curr := head
	var pre *ListNode
	for curr != nil {
		next := curr.Next
		curr.Next = pre
		pre = curr
		curr = next
	}
```

#### 25 K个一组反转链表
```go
func reverseKGroup(head *ListNode, k int) *ListNode {
    if head == nil || head.Next == nil || k == 1 {
        return head
    }
    prev := &ListNode{0, head}
    pre := prev
    slow, fast := head, head
    for i := 0; i < k - 1; i++ {
        fast = fast.Next
    }
    count := 0
    for fast != nil {
        if count % k == 0 {
            slow, fast = reverse(slow, fast)
            pre.Next = slow
        }
        pre = pre.Next
        slow = slow.Next
        fast = fast.Next
        count++
    }
    return prev.Next
}
func reverse(head, tail *ListNode) (*ListNode, *ListNode) {
    prev := tail.Next
    cur := head
    for prev != tail {
        next := cur.Next
        cur.Next = prev
        prev = cur
        cur = next
    }
    return tail, head
}
```

```go
func mergeTwoLists(l1 *ListNode, l2 *ListNode) *ListNode {
    prev := &ListNode{}
    cur := prev
    for l1 != nil && l2 != nil {
        if l1.Val < l2.Val {
            next := l1.Next
            cur.Next = l1
            l1.Next = nil
            l1 = next
        } else {
            next := l2.Next
            cur.Next = l2
            l2.Next = nil

        }
        cur = cur.Next
    }
    if l1 != nil {
        cur.Next = l1
    }
    if l2 != nil {
        cur.Next = l2
    }
    return prev.Next
}
```

```go
func getIntersectionNode(headA, headB *ListNode) *ListNode {
    p1, p2 := headA, headB
    for p1 != p2 {
        if p1 == nil {
            p1 = headB
        } else {
            p1 = p1.Next
        }
        if p2 == nil {
            p2 = headA
        } else {
            p2 = p2.Next
        }
    }
```

```go
func hasCycle(head *ListNode) bool {
    if head == nil || head.Next == nil {
        return false
    }
    slow, fast := head, head.Next
    for slow != fast {
        if fast == nil || fast.Next == nil {
            return false
        }
        slow = slow.Next
        fast = fast.Next.Next
    }
    return true
}
```

```go
func detectCycle(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return nil
    }
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
        if slow == fast {
            slow = head
            for slow != fast {
                slow = slow.Next
                fast = fast.Next
            }
            return slow
        }
    }
    return nil
}
```

```go
        fast = fast.Next
```

```go
func isPalindrome(head *ListNode) bool {
    if head == nil || head.Next == nil {
        return true
    }
    var pre *ListNode
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        pre = slow
        slow = slow.Next
        fast = fast.Next.Next
    }
    pre.Next = nil
    // 反转后一段链表
    var head2 *ListNode
    for slow != nil {
        next := slow.Next
        slow.Next = head2
        head2 = slow
        slow = next
    }
    // 比较两段链表
    for head != nil {
        if head.Val != head2.Val {
            return false
        }
        head = head.Next
        head2 = head2.Next
    }
    return true
}
```

```go
func deleteDuplicates(head *ListNode) *ListNode {
    if head == nil {
        return nil
    }
    cur := head
    for cur.Next != nil {
        if cur.Val == cur.Next.Val {
            cur.Next = cur.Next.Next
        } else {
            cur = cur.Next
        }
    }
    return head
}
```

```go
func deleteDuplicates(head *ListNode) *ListNode {
    if head == nil {
        return nil
    }
    prev := &ListNode{0, head}
    cur := prev
    for cur.Next != nil && cur.Next.Next != nil {
        // 如果后面两个元素相等，就都删掉
        if cur.Next.Val == cur.Next.Next.Val {
            x := cur.Next.Val
            for cur.Next != nil && cur.Next.Val == x {
                cur.Next = cur.Next.Next
            }
        } else {
            cur = cur.Next
        }
    }
    return prev.Next
}
```

#### 24 两两交换链表中的节点
```go
func swapPairs(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }
    prev := &ListNode{0, head}
    cur := prev
    // 迭代法，两两交换节点
    for cur.Next != nil && cur.Next.Next != nil {
        node1 := cur.Next
        node2 := cur.Next.Next
        cur.Next = node2
        node1.Next = node2.Next
        node2.Next = node1
        cur = node1
    }
    return prev.Next
}
```

```go
func mergeKLists(lists []*ListNode) *ListNode {
    if len(lists) == 0 {
        return nil
    }
    if len(lists) == 1 {
        return lists[0]
    }
    l := mergeTwoLists(lists[0], lists[1])
    for i := 2; i < len(lists); i++ {
        l = mergeTwoLists(l, lists[i])
    }
    return l
```
