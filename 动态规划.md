```go
func longestPalindrome(s string) string {
    var res string
    dp := make([][]bool, len(s))
    for i := 0; i < len(dp); i++ {
        dp[i] = make([]bool, len(s))
    }
    for l := 0; l < len(s); l++ {
        for i := 0; i + l < len(s); i++ {
            j := i + l
            if l == 0 {
                dp[i][j] = true
            } else if l == 1 {
                dp[i][j] = (s[i] == s[j])
            } else {
                dp[i][j] = (s[i] == s[j]) && dp[i + 1][j - 1]
            }
            if dp[i][j] && len(res) < l + 1 {
                res = s[i:j+1]
            }
        }
    }
    return res
}
```

#### 300 最长递增子序列
```go
func lengthOfLIS(nums []int) int {
	// dp[i]表示下标为i的数组含有的最长递增子序列
	var maxLength int
	dp := make([]int, len(nums))
	for i := 0; i < len(nums); i++ {
		dp[i] = 1
		// 内层维护dp数组的最大值
		for j := 0; j < i; j++ {
			if nums[j] < nums[i] && dp[i] < dp[j] + 1 {
				dp[i] = dp[j] + 1
			}
		}
		if dp[i] > maxLength {
			maxLength = dp[i]
		}
	}
	return maxLength
}
```

```go
        return n
    }
    f1, f2, f3 := 1, 2, 0
    for i := 3; i <= n; i++ {
        f3 = f1 + f2
```

```go
func fib(n int) int {
    if n <= 1 {
        return n
    }
    f0, f1, f2 := 0, 1, 0
    for i := 2; i <= n; i++ {
        f2 = (f0 + f1) % 1000000007
        f0 = f1
        f1 = f2
    }
    return f2 % 1000000007
}
```

```go
func longestCommonSubsequence(text1 string, text2 string) int {
    m, n := len(text1), len(text2)
    dp := make([][]int, m + 1)
    for i := 0; i < len(dp); i++ {
        dp[i] = make([]int, n + 1)
    }
    for i := 1; i <= m; i++ {
        for j := 1; j <= n; j++ {
            dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
            if text1[i - 1] == text2[j - 1] {
                dp[i][j] = max(dp[i][j], dp[i - 1][j - 1] + 1)
            }
        }
    }
    return dp[m][n]
}
```

```go
func findLength(nums1 []int, nums2 []int) int {
    var res int
    m, n := len(nums1), len(nums2)
    dp := make([][]int, m + 1)
    for i := 0; i <= m; i++ {
        dp[i] = make([]int, n + 1)
    }
    for i := 1; i <= m; i++ {
        for j := 1; j <= n; j++ {
            if nums1[i - 1] == nums2[j - 1] {
                dp[i][j] = dp[i - 1][j - 1] + 1
            } else {
                dp[i][j] = 0
            }
            if dp[i][j] > res {
                res = dp[i][j]
            }
        }
    }
    return res
}
```

```go
func rob(nums []int) int {
    if len(nums) == 1 {
        return nums[0]
    }
    dp := make([]int, len(nums))
    dp[0] = nums[0]
    dp[1] = max(nums[0], nums[1])
    for i := 2; i < len(nums); i++ {
        dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])
    }
    return dp[len(nums) - 1]
}
```

```go
func minPathSum(grid [][]int) int {
    m, n := len(grid), len(grid[0])
    dp := make([][]int, m)
    dp[0] = make([]int, n)
    dp[0][0] = grid[0][0]
    for i := 1; i < m; i++ {
        dp[i] = make([]int, n)
        dp[i][0] = grid[i][0]
        dp[i][0] += dp[i - 1][0]
    }
    for i := 1; i < n; i++ {
        dp[0][i] = grid[0][i]
        dp[0][i] += dp[0][i - 1]
    }
    for i := 1; i < m; i++ {
        for j := 1; j < n; j++ {
            dp[i][j] = grid[i][j]
            dp[i][j] += min(dp[i - 1][j], dp[i][j - 1])
        }
    }
    return dp[m - 1][n - 1]
}
```

```go
func coinChange(coins []int, amount int) int {
    dp := make([]int, amount + 1)
    for i, _ := range dp {
        dp[i] = math.MaxInt32
    }
    dp[0] = 0
    for i := 0; i < len(coins); i++ {
        for j := coins[i]; j <= amount; j++ {
            if dp[j - coins[i]] != math.MaxInt32 {
                dp[j] = min(dp[j], dp[j - coins[i]] + 1)
            }
        }
    }
    if dp[amount] == math.MaxInt32 {
        return -1
    }
    return dp[amount]
}
```

```go
func change(amount int, coins []int) int {
	// dp[i]表示凑成面额为i有dp[i]种组合
	dp := make([]int, amount + 1)
	// 初始化，组成面额为0的有一种
	dp[0] = 1
	for i := 0; i < len(coins); i++ {
		for j := coins[i]; j <= amount; j++ {
			// 统计面额为j时，有几种组合
			dp[j] += dp[j - coins[i]]
		}
	}
	return dp[amount]
}
```

```go
func uniquePaths(m int, n int) int {
    dp := make([][]int, m)
    for i := 0; i < m; i++ {
        dp[i] = make([]int, n)
        dp[i][0] = 1
    }
    for i := 0; i < n; i++ {
        dp[0][i] = 1
    }
    for i := 1; i < m; i++ {
        for j := 1; j < n; j++ {
            dp[i][j] += dp[i - 1][j] + dp[i][j - 1]
        }
    }
    return dp[m - 1][n - 1]
}
```

#### 72	编辑距离
```go
func minDistance(word1 string, word2 string) int {
	// 可以想象成3种操作
	//1. 在单词 A 中插入一个字符
	//2. 在单词 B 中插入一个字符
	//3. 修改单词 A 的一个字符
	// dp[i][j]表示长度为i的字符串变成长度为j的字符串需要的最少操作数
	dp := make([][]int, len(word1) + 1)
	for i := 0; i < len(dp); i++ {
		dp[i] = make([]int, len(word2) + 1)
		dp[i][0] = i
	}
	for i := 0; i < len(dp[0]); i++ {
		dp[0][i] = i
	}

	for i := 1; i < len(dp); i++ {
		for j := 1; j < len(dp[0]); j++ {
			// word1[i - 1]不等于word2[j - 1]
			dp[i][j] = min(min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1
			// word1[i - 1]等于word2[j - 1]
			if word1[i - 1] == word2[j - 1] {
				dp[i][j] = min(dp[i][j], dp[i - 1][j - 1])
			}
		}
	}
	return dp[len(dp) - 1][len(dp[0]) - 1]
}
```

#### 96	不同的二叉搜索树
```go
func numTrees(n int) int {
	// dp[i]表示节点个数为n一共有几种情况
	dp := make([]int, n+1)
	dp[0], dp[1] = 1, 1
	// 外层遍历节点个数
	for i := 2; i <= n; i++ {
		// 内层遍历以j为根节点，且节点个数为i的情况
		for j := 1; j <= i; j++ {
			dp[i] += dp[j-1] * dp[i-j]
		}
	}
	return dp[n]
}
```

```go
func wordBreak(s string, wordDict []string) bool {
	// 初始化物品
	hashTable := make(map[string]bool)
	for _, v := range wordDict {
		hashTable[v] = true
	}
	// dp[i] 表示i之前的字符串都能从字典中拼接出来
	dp := make([]bool, len(s) + 1)
	dp[0] = true
	// 外层遍历背包
	for i := 0; i <= len(s); i++ {
		// 内层遍历物品
		for j := 0; j < i; j++ {
			if dp[j] && hashTable[s[j:i]] {
				dp[i] = true
			}
		}
	}
	return dp[len(s)]
}
```

#### 221 最大正方形
```go
func maximalSquare(matrix [][]byte) int {
	// 保存最大边长
	var maxSide int
	// dp[i][j]表示以matrix[i][j]为右下角顶点正方形的最大边长
	dp := make([][]int, len(matrix))
	// 初始化
	for i := 0; i < len(matrix); i++ {
		dp[i] = make([]int, len(matrix[i]))
		for j := 0; j < len(matrix[i]); j++ {
			dp[i][j] = int(matrix[i][j] - '0')
			if dp[i][j] == 1 {
				maxSide = 1
			}
		}
	}
	for i := 1; i < len(matrix); i++ {
		for j := 1; j < len(matrix[i]); j++ {
			// 如果当前节点为1，最大边长等于相邻三个位置元素的最小值加1
			if dp[i][j] == 1 {
				dp[i][j] = min(min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1
			}
			// 更新最大边长
			if dp[i][j] > maxSide {
				maxSide = dp[i][j]
			}
		}
	}
	return maxSide * maxSide
}

func min(a, b int) int {
	if a < b {
		return a
	} else {
		return b
	}
}
```
