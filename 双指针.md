#### 3 无重复字符的最长子串
```go
func lengthOfLongestSubstring(s string) int {
    // 用于存储每个元素的下一个位置，方便左指针快速跳转
    indexMap := make([]int, 128)
    var res int
    l := 0
    for i := 0; i < len(s); i++ {
        // 当前元素的下一个位置比l大，说明左右指针之间存在重复元素
        // 左指针移动到前一个重复元素的下一个位置
        if indexMap[s[i]] > l {
            // 跳转到重复元素的下一个位置
            l = indexMap[s[i]]
        }
        // 更新结果值
        if i - l + 1 > res {
            res = i - l + 1
        }
        // 更新当前元素的下一个位置
        indexMap[s[i]] = i + 1
    }
    return res
}
```

#### 53	最大子数组和
```go
func maxSubArray(nums []int) int {
    pre, maxSum := 0, math.MinInt32
    for i := 0; i < len(nums); i++ {
        if pre + nums[i] > nums[i] {
            pre += nums[i]
        } else {
            pre = nums[i]
        }
        if pre > maxSum {
            maxSum = pre
        }
    }
    return maxSum
}
```

#### 15	三数之和
```go
func threeSum(nums []int) [][]int {
    var res [][]int
    sort.Ints(nums)
    for i := 0; i < len(nums) - 2; i++ {
        // 剪枝
        if nums[i] > 0 {
            break
        }
        // 去重
        if i != 0 && nums[i] == nums[i - 1] {
            continue
        }
        // 初始化中指针和右指针
        j, k := i + 1, len(nums) - 1
        for j < k {
            if nums[i] + nums[j] + nums[k] == 0 {
                res = append(res, []int{nums[i], nums[j], nums[k]})
                // 对j和k去重
                n1, n2 := nums[j], nums[k]
                for j < k && nums[j] == n1 {
                    j++
                }
                for j < k && nums[k] == n2 {
                    k--
                }
            } else if nums[i] + nums[j] + nums[k] > 0 {
                k--
            } else {
                j++
            }
        }
    }
    return res
}
```

#### 16 最接近的三数之和

```go
func threeSumClosest(nums []int, target int) int {
    sort.Ints(nums)
    res := nums[0] + nums[1] + nums[2]
    for i := 0; i < len(nums) - 2; i++ {
        if i != 0 && nums[i] == nums[i - 1] {
            continue
        }
        j, k := i + 1, len(nums) - 1
        for j < k {
            sum := nums[i] + nums[j] + nums[k]
            if sum == target {
                return sum
            } else if sum > target {
                k--
            } else {
                j++
            }
            if abs(sum - target) < abs(res - target) {
                res = sum
            }
        }
    }
    return res
}
func abs(a int) int {

        return -a
    }
    return a
}
```

#### 121 买卖股票的最佳时机
```go
func maxProfit(prices []int) int {
    maxValue, minValue := 0, math.MaxInt32
    for i := 0; i < len(prices); i++ {
        if prices[i] < minValue {
            minValue = prices[i]
        } else if prices[i] - minValue > maxValue {
            maxValue = prices[i] - minValue
        }
    }
    return maxValue
}
```

#### 88	合并两个有序数组

```go
func merge(nums1 []int, m int, nums2 []int, n int)  {
    i, j, k := m - 1, n - 1, len(nums1) - 1
    for i >= 0 && j >= 0 {
        if nums1[i] > nums2[j] {
            nums1[k] = nums1[i]
            i--
        } else {
            nums1[k] = nums2[j]
            j--
        }
        k--
    }
    for j >= 0 {
        nums1[k] = nums2[j]
        k--
        j--
    }
}
```

#### 42	接雨水
```go
func trap(height []int) int {
    var res int
    l, r := 0, len(height) - 1
    for l < r {
        if height[l] < height[r] {
            if height[l] > height[l + 1] {
                res += height[l] - height[l + 1]
                height[l + 1] = height[l]
            }
            l++
        } else {
            if height[r] > height[r - 1] {
                res += height[r] - height[r - 1]
                height[r - 1] = height[r]
            }
            r--
        }
    }
    return res
}
```

```go
func exchange(nums []int) []int {
    slow, fast := 0, 0
    for _, v := range nums {
        if v % 2 == 1 {
            nums[slow], nums[fast] = nums[fast], nums[slow]
            slow++
        }
        fast++
    }
    return nums
}
```

#### 209 长度最小的子数组
```go
func minSubArrayLen(target int, nums []int) int {
	n := len(nums)
	// 初始化最小值
	ans := math.MaxInt32
	slow, fast := 0, 0
	// 用于累加和
	sum := 0
	for fast < n {
		sum += nums[fast]
		// target大于等于target，slow指针开始向后移动
		for sum >= target {
			// 更新最小长度
			if fast - slow + 1 < ans {
				ans = fast - slow + 1
			}
			// 减去最左边的值
			sum -= nums[slow]
			// 右移慢指针
			slow++
		}
		// 后移快指针
		fast++
	}
	// 如果没有满足的子数组，就返回0
	if ans == math.MaxInt32 {
		return 0
	}
	return ans
}
```
