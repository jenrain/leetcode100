#### 912	排序数组
```go
func sortArray(nums []int) []int {
    var quicksort func(l, r int, nums []int)
    quicksort = func(l, r int, nums []int) {
        i, j := l, r
        // 将当前序列在中间位的数定义为分隔数
        mid := nums[(i + j) / 2]
        for i <= j {
            // 扫描左半部分比分隔数大的数
            for nums[i] < mid {
                i++
            }
            // 扫描右半部分比分隔数小的数
            for nums[j] > mid {
                j--
            }
            // 交换上述找到的两个数
            if i <= j {
                nums[i], nums[j] = nums[j], nums[i]
                i++
                j--
            }
        }
        // 递归对左右两边序列继续排序
        if i < r {
            quicksort(i, r, nums)
        }
        if l < j {
            quicksort(l, j, nums)
        }
    }
    quicksort(0, len(nums) - 1, nums)
    return nums
}
```
