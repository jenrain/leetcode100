```go
func permute(nums []int) [][]int {
    var res [][]int
    visited := make(map[int]bool)
    var dfs func(path []int)
    dfs = func(path []int) {
        if len(path) == len(nums) {
            res = append(res, append([]int{}, path...))
            return
        }
        for i := 0; i < len(nums); i++ {
            if visited[nums[i]] {
                continue
            }
            visited[nums[i]] = true
            dfs(append(path, nums[i]))
            visited[nums[i]] = false
        }
    }
    dfs([]int{})
    return res
}
```

#### 200 岛屿数量
```go
func numIslands(grid [][]byte) int {
    var res int
    for i := 0; i < len(grid); i++ {
        for j := 0; j < len(grid[0]); j++ {
            if grid[i][j] == '1' {
                res++
                dfs(grid, i, j)
            }
        }
    }
    return res
}

func dfs(grid [][]byte, i, j int) {
    if i < 0 || i >= len(grid) || j < 0 || j >= len(grid[0]) {
        return
    }
    if grid[i][j] == '0' {
        return
    }
    grid[i][j] = '0'
    dfs(grid, i - 1, j)
    dfs(grid, i, j - 1)
    dfs(grid, i + 1, j)
    dfs(grid, i, j + 1)
}
```

#### 39	组合总和
```go
func combinationSum(candidates []int, target int) [][]int {
    var res [][]int
    var dfs func(i, target int, path []int)
    dfs = func(i, target int, path []int) {
        if target == 0 {
            res = append(res, append([]int{}, path...))
            return
        }
        for j := i; j < len(candidates); j++ {
            if target >= candidates[j] {
                dfs(j, target - candidates[j], append(path, candidates[j]))
            }
        }
    }
    dfs(0, target, []int{})
    return res
}
```

#### 78	子集

```go
func subsets(nums []int) [][]int {
    var res [][]int
    var dfs func(i int, path []int)
    dfs = func(i int, path []int) {
        if i == len(nums) {
            res = append(res, append([]int{}, path...))
            return
        }
        dfs(i + 1, path)
        dfs(i + 1, append(path, nums[i]))
    }
    dfs(0, []int{})
    return res
}
```

#### 79	单词搜索
```go
func exist(board [][]byte, word string) bool {
	type pair struct {
		x, y int
	}
	m, n := len(board), len(board[0])
	dirs := []pair{{-1,0}, {0, -1}, {1, 0}, {0,1}}
	visited := make([][]bool, m)
	for i := 0; i < m; i++ {
		visited[i] = make([]bool, n)
	}
	var dfs func(i, j, k int) bool
	dfs = func(i, j, k int) bool {
		// 剪枝
		if word[k] != board[i][j] {
			return false
		}
		// 匹配成功
		if k == len(word) - 1 {
			return true
		}
		// 标记
		visited[i][j] = true
		// 回溯
		defer func() {
			visited[i][j] = false
		}()
		// 遍历四个方向
		for _, dir := range dirs {
			newX, newY := i + dir.x, j + dir.y
			if newX >= 0 && newX < m && newY >= 0 && newY < n && !visited[newX][newY] {
				if dfs(newX, newY, k + 1) {
					return true
				}
			}
		}
		return false
	}
	for i := 0; i < m; i++ {
		for j := 0; j < n; j++ {
			if dfs(i, j, 0) {
				return true
			}
		}
	}
	return false
}
```
#### 77	组合
```go
func combine(n int, k int) [][]int {
	var res [][]int
	var dfs func(start int, path []int)
	dfs = func(start int, path []int) {
		if k == len(path) {
			res = append(res, append([]int{}, path...))
			return
		}
		for i := start; i <= n; i++ {
			dfs(i + 1, append(path, i))
		}
	}
	dfs(1, []int{})
	return res
}
```
